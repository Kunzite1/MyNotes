# 事件集实现键控灯

## 引脚、外设、模块

这次用到两个按键、一个LED

引脚：

1. KEY1 -> PD9

2. KEY2 -> PD8

3. LED1 -> PA0

4. LED2 -> PA1

5. LED3 -> PA2

引脚初始化放在一个专门的函数pin_init()里

```c
void pin_init()
{
    //设置IO口模式
    rt_pin_mode(KEY1, PIN_MODE_INPUT_PULLUP);
    rt_pin_mode(KEY2, PIN_MODE_INPUT_PULLUP);
    rt_pin_mode(LED1, PIN_MODE_OUTPUT);
    rt_pin_mode(LED2, PIN_MODE_OUTPUT);
    rt_pin_mode(LED3, PIN_MODE_OUTPUT);
}
```

## 一、创建事件和线程

1. 像这样，定义一个事件

```c
rt_event_t LED_event;
```

然后在main函数里用`rt_event_creat()`来创建事件

```c
LED_event = rt_event_create("LED", RT_IPC_FLAG_FIFO);
```

第二个参数，是事件集的运行规则，这里用的是<u>FIFO：先进先出</u>。

2. 创建线程

跟[流水灯](流水灯.md)相同，两个线程：LED和KEY

## 二、事件集的使用

### 1. 发送事件集

在KEY线程的入口函数中发送事件集，使用事件发送函数`re_event_sent()`

```c
void KEY_entry(void *p)
{
    unsigned char key_num = 0, key_down=0, key_old=0;
    while(1)
    {
        if(rt_pin_read(KEY1) == 0)
            key_num = 1;
        else if (rt_pin_read(KEY2) == 0)
            key_num = 2;
        else
            key_num = 0;

        //视频教程里写的按键消抖
        key_down = key_num & (key_num^key_old);
        key_old = key_num;

        if (key_down == 1)
            rt_kprintf("key1 YES\n");
        else
            rt_kprintf("key1 NO\n");

        //按键处理
        switch (key_down)
        case 1: rt_event_send(LED_event, LOC(0)); break;

    }

    rt_thread_mdelay(15);
}
```

### 解释一下LOC(i)

`#define LOC(i) 0x01<<i`

用宏定义使LOC(i)表示0X01左移i位的结果。

所以上面代码`rt_event_send(LED_event, LOC(0));`表示发送了LED_event事件集中的事件0

### 2. 接收事件集

在LED线程的入口函数里接收事件，使用接收函数`rt_event_recv()`

```c
void LED_entry(void *p)
{
    unsigned char result = 0;
    while(1)
    {
        rt_event_recv(LED_event, LOC(0), RT_EVENT_FLAG_AND | RT_EVENT_FLAG_CLEAR, RT_WAITING_FOREVER, 0);
        result = !result;
        rt_pin_write(LED1, result);

        rt_kprintf("result=%d", result);
    }
}
```

## 三、事件集键控灯工作原理

当按键按下，则发送一个事件，在LED线程中，由于接收函数的等待时间设为FOREVER，LED函数会一直停留于等待接收。当接收到按键触发的事件发送信号后，LED线程才开始执行后续动作，即result取反，LED亮灭转换。

```c
rt_event_recv(LED_event, LOC(0), RT_EVENT_FLAG_AND | RT_EVENT_FLAG_CLEAR, RT_WAITING_FOREVER, 0);
```

## 本次工程main.c代码

代码中加了一个按键KEY2，任意按下一个按键均可使LED亮灭转换。

```c
#include <rtthread.h>

#define DBG_TAG "main"
#define DBG_LVL DBG_LOG
#include <rtdbg.h>

#include <rtdevice.h>
#include <board.h>

#define LED1 GET_PIN(A, 0)
#define LED2 GET_PIN(A, 1)
#define LED3 GET_PIN(A, 2)
#define KEY1 GET_PIN(D, 9)
#define KEY2 GET_PIN(D, 8)      //定义按键和LED引脚

#define LOC(i) 0x01<<i  //宏定义，表示0x01左移i位后的值

//定义事件和线程
rt_thread_t LED, KEY;
rt_event_t LED_event;

void pin_init()
{
    //设置IO口模式
    rt_pin_mode(KEY1, PIN_MODE_INPUT_PULLUP);
    rt_pin_mode(KEY2, PIN_MODE_INPUT_PULLUP);
    rt_pin_mode(LED1, PIN_MODE_OUTPUT);
    rt_pin_mode(LED2, PIN_MODE_OUTPUT);
    rt_pin_mode(LED3, PIN_MODE_OUTPUT);
}

//线程运行函数
void KEY_entry(void *p)
{
    unsigned char key_num = 0, key_down=0, key_old=0;
    while(1)
    {
        if(rt_pin_read(KEY1) == 0)
            key_num = 1;
        else if (rt_pin_read(KEY2) == 0)
            key_num = 2;
        else
            key_num = 0;

        //视频教程里写的按键消抖
        key_down = key_num & (key_num^key_old);
        key_old = key_num;

        if (key_down == 1)
            rt_kprintf("key1 YES\n");

        //按键处理
        switch (key_down)
        {
        //发送事件
        case 1: rt_event_send(LED_event, LOC(0)); break;
        case 2: rt_event_send(LED_event, LOC(1)); break;
        }

        rt_thread_mdelay(15);
    }
}

void LED_entry(void *p)
{
    unsigned char result = 0;
    while(1)
    {
        //接收事件
        rt_event_recv(LED_event, LOC(0)|LOC(1), RT_EVENT_FLAG_OR | RT_EVENT_FLAG_CLEAR, RT_WAITING_FOREVER, 0);
        //LED亮灭转换
        result = !result;
        rt_pin_write(LED1, result);
    }
}

int main(void)
{
    //IO口初始化
    pin_init();
    //创建线程
    KEY = rt_thread_create("KEY", KEY_entry, RT_NULL, 512, 13, 5);
    LED = rt_thread_create("LED", LED_entry, RT_NULL, 512, 15, 5);
    //创建事件
    LED_event = rt_event_create("LED", RT_IPC_FLAG_FIFO);   //FIFO：先进先出模式

    if (KEY != RT_NULL)
        rt_thread_startup(KEY);
    else
        rt_kprintf("KEY failed \n");

    if (LED != RT_NULL)
        rt_thread_startup(LED);
    else
        rt_kprintf("LED failed \n");

    return RT_EOK;
}
```
