# 7 互斥量

## 7-1 简介

互斥信号量就是一个拥有优先级继承的二值信号量。二值信号量常用于任务或中断之间的同步，互斥信号量常用于那些需要互斥访问的应用中。

- 要使用互斥量，需要在FreeRTOSConfig.h中把宏`configUSE_MUTEXES`定义为1.

- 要使用递归互斥量，需要在FreeRTOSConfig.h中把宏`configUSE_RECURSIVE_MUTEXES`定义为1.

## 7-2 优先级继承机制

假设有三个任务ABC，A优先级最高，C优先级最低。

C正在访问一个互斥信号量。当A想访问此互斥量时，A会进入阻塞状态，同时优先级继承算法会把C的优先级，暂时提高到A的级别，等C释放互斥量时，优先级回到初始值。

这样就能避免优先级翻转的情况（高优先级等待低优先级任务时，CPU被中优先级的任务抢占），即减少高优先级任务的等待/阻塞时间。

## 7-2 应用场景

应用于可能产生优先级翻转的情况。

## 7-3 常用API函数

### 7-3-1 创建函数

- `xSemaphoreCreateMutex()`常用互斥量创建函数

- `xSemaphoreCreateRecursiveMutex()`创建递归互斥量，返回值是递归互斥量句柄，创建失败则返回NULL。

### 7-3-2 删除函数

- `xSemaphoreDelete()`

### 7-3-3 获取函数

- `xSemaphoreTake()`获取互斥量，函数使用与普通信号量一致，使用时传入句柄有区别，同时，互斥量具有优先级继承机制。
- `xSemaphoreTakeRecursive()`递归互斥量获取函数。 

### 7-3-4 释放函数

- `xSemaphoreGive()`

## 7-4 示例

```c
#include "system.h"
#include "SysTick.h"
#include "led.h"
#include "usart.h"
#include "FreeRTOS.h"
#include "task.h"
#include "key.h"
#include "semphr.h"

// 切换二值信号量和互斥量，宏定义为1使用互斥量，为0使用二值信号量
// 先使用二值信号量可以观察到优先级翻转的现象
// 然后使用互斥量，互斥量的优先级继承机制能避免优先级翻转的现象。
#define BINARY_OR_MUTEX     1

//任务优先级
#define START_TASK_PRIO        1
//任务堆栈大小    
#define START_STK_SIZE         128  
//任务句柄
TaskHandle_t StartTask_Handler;
//任务函数
void start_task(void *pvParameters);

// LED1任务
#define LED1_TASK_PRIO        2
#define LED1_STK_SIZE         50  
TaskHandle_t LED1Task_Handler;
void led1_task(void *pvParameters);

// 低优先级任务
#define LowPrio_TASK_PRIO   3
#define LowPrio_STK_SIZE    512
TaskHandle_t LowPrio_Task_Handler;
void LowPrio_task(void *pvParameters);

// 中优先级任务
#define MidPrio_TASK_PRIO   4
#define MidPrio_STK_SIZE    512
TaskHandle_t MidPrio_Task_Handler;
void MidPrio_task(void *pvParameters);

// 高优先级任务
#define HighPrio_TASK_PRIO   5
#define HighPrio_STK_SIZE    512
TaskHandle_t HighPrio_Task_Handler;
void HighPrio_task(void *pvParameters);

#if (BINARY_OR_MUTEX == 0)
// 二值信号量句柄
SemaphoreHandle_t BinarySem_Handle = NULL;
#endif

#if (BINARY_OR_MUTEX == 1)
// 互斥信号量句柄
SemaphoreHandle_t MutexSem_Handle = NULL;
#endif

int main()
{
    SysTick_Init(72);
    NVIC_PriorityGroupConfig(NVIC_PriorityGroup_4);//设置系统中断优先级分组4
    LED_Init();
    USART1_Init(115200);
    printf("Program starting!\n");

    //创建开始任务
    xTaskCreate((TaskFunction_t )start_task,            //任务函数
                (const char*    )"start_task",          //任务名称
                (uint16_t       )START_STK_SIZE,        //任务堆栈大小
                (void*          )NULL,                  //传递给任务函数的参数
                (UBaseType_t    )START_TASK_PRIO,       //任务优先级
                (TaskHandle_t*  )&StartTask_Handler);   //任务句柄              
    vTaskStartScheduler();          //开启任务调度
}

//开始任务任务函数
void start_task(void *pvParameters)
{
    taskENTER_CRITICAL();           //进入临界区

    #if (BINARY_OR_MUTEX == 0)
    BaseType_t xReturn = pdPASS;

    // 创建二值信号量
    BinarySem_Handle = xSemaphoreCreateBinary();
    xReturn = xSemaphoreGive(BinarySem_Handle);
    #endif

    #if (BINARY_OR_MUTEX == 1)
    // 创建互斥量
    MutexSem_Handle = xSemaphoreCreateMutex();
    #endif

    //创建LED1任务
    xTaskCreate((TaskFunction_t )led1_task,     
                (const char*    )"led1_task",   
                (uint16_t       )LED1_STK_SIZE, 
                (void*          )NULL,
                (UBaseType_t    )LED1_TASK_PRIO,
                (TaskHandle_t*  )&LED1Task_Handler); 
    // 创建LowPrio任务
    xTaskCreate((TaskFunction_t )LowPrio_task,     
                (const char*    )"LowPrio_task",   
                (uint16_t       )LowPrio_STK_SIZE, 
                (void*          )NULL,
                (UBaseType_t    )LowPrio_TASK_PRIO,
                (TaskHandle_t*  )&LowPrio_Task_Handler); 
    // 创建MidPrio任务
    xTaskCreate((TaskFunction_t )MidPrio_task,     
                (const char*    )"MidPrio_task",   
                (uint16_t       )MidPrio_STK_SIZE, 
                (void*          )NULL,
                (UBaseType_t    )MidPrio_TASK_PRIO,
                (TaskHandle_t*  )&MidPrio_Task_Handler); 
    // 创建HighPrio任务
    xTaskCreate((TaskFunction_t )HighPrio_task,     
                (const char*    )"HighPrio_task",   
                (uint16_t       )HighPrio_STK_SIZE, 
                (void*          )NULL,
                (UBaseType_t    )HighPrio_TASK_PRIO,
                (TaskHandle_t*  )&HighPrio_Task_Handler); 

    vTaskDelete(StartTask_Handler); //删除开始任务
    taskEXIT_CRITICAL();            //退出临界区
} 

//LED1任务函数
void led1_task(void *pvParameters)
{
    while(1)
    {
        LED_Ctrl(1, LIGHT_ON);
        vTaskDelay(800);
        LED_Ctrl(1, LIGHT_OFF);
        vTaskDelay(200);
    }
}

#if (BINARY_OR_MUTEX == 1)
// 低优先级任务
void LowPrio_task(void *pvParameters)
{
    BaseType_t xReturn = pdTRUE;
    static int i = 0;

    while(1)
    {
        printf("LLL: Getting MutexSem\n");
        xReturn = xSemaphoreTake(MutexSem_Handle, portMAX_DELAY);

        if (xReturn == pdTRUE) {
            printf("LLL: Successfully running\n");
            LED_toggle();

            for (i = 0; i < 2000000; i++) {
                taskYIELD();
            }
        }

        printf("LLL: Give MutexSem\n");
        xReturn = xSemaphoreGive(MutexSem_Handle);
        vTaskDelay(500);
    }
}

// 中优先级任务
void MidPrio_task(void *pvParameters)
{
    while(1)
    {
        printf("MMM: Mid is running\n");
        vTaskDelay(500);
    }
}

// 高优先级任务
void HighPrio_task(void *pvParameters)
{
    BaseType_t xReturn = pdTRUE;

    while(1)
    {
        // 获取二值信号量
        printf("HHH: Getting MutexSem!\n");
        xReturn = xSemaphoreTake(MutexSem_Handle, portMAX_DELAY);

        if (xReturn == pdTRUE) {
            printf("HHH: Successfully running1\n");
            LED_toggle();
        }

        printf("HHH: Give MutexSem\n");
        xReturn = xSemaphoreGive(MutexSem_Handle);
        vTaskDelay(500);
    }
}
#endif
```
